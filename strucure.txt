my-ts-blockchain/
├── index.html                   # Точка входа для браузера (если потребуется 
визуализация)
├── package.json                 # Зависимости и скрипты запуска
├── tsconfig.json                # Конфигурация TypeScript
├── .gitignore                   # Исключения Git
├── src/                         # Исходный код проекта
│   ├── index.ts                 # Точка входа (запускает и тестирует блокчейн)
│   ├── core/                    # Основная логика блокчейна
│   │   ├── Block.ts             # Модель блока (уже есть)
│   │   └── Blockchain.ts        # Логика цепочки блоков (создание, добавление, 
проверка)
│   ├── types/                   # Общие интерфейсы и типы
│   │   └── Transaction.ts       # Структура транзакции (напр., sender, receiver, 
amount)
│   ├── crypto/                  # Криптографические утилиты
│   │   └── hash.ts              # Функции хеширования (можно будет вынести 
calculateHash сюда)
│   ├── wallet/                  # Работа с ключами и подписями
│   │   ├── Wallet.ts            # Кошелёк пользователя (генерация ключей, подпись)
│   │   └── Signature.ts         # Подпись транзакций (ECDSA и пр.)
│   └── utils/                   # Вспомогательные функции
│       └── formatter.ts         # Форматирование вывода, логгеры и т.д.
└── public/                      # Статика (если нужно подключать фронтенд)
    └── style.css                # Будущие стили интерфейса
